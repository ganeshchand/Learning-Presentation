Make folds, not loops.

-- Sum numbers from 1 to n the Haskell way:
    sumTo n = foldl1 (+) [1..n]

Weird Scala operator for folding

// Three ways to fold in Scala:
    def sumTo0(n: Int) = (1 to n).foldLeft(0) { (x, y) => x + y }
    def sumTo1(n: Int) = (1 to n).foldLeft(0) { _ + _ }
    def sumTo2(n: Int) = (0 /: (1 to n)) { _ + _ }
